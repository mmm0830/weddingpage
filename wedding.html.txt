// ★ここをあとで変える
const correct = ["え","が","お","だ","よ"];

function hira(str){
  return str.trim().replace(/[ァ-ン]/g,s =>
    String.fromCharCode(s.charCodeAt(0)-0x60)
  );
}

// 自動で次のマスへ
const ids = ["c1","c2","c3","c4","c5"];

ids.forEach((id,i)=>{
  document.getElementById(id).addEventListener("input",()=>{
    if(i < ids.length-1){
      document.getElementById(ids[i+1]).focus();
    }
  });
});

function checkAnswer(){

  const input = ids.map(id =>
    hira(document.getElementById(id).value)
  );

  if(JSON.stringify(input) === JSON.stringify(correct)){

    location.href = "ending.html";

  }else{

    document.getElementById("message").innerHTML =
      "ちがうみたい…";

  }

}
